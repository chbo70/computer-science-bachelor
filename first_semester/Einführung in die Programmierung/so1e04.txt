#include <stdio.h> //bindet die Header-Datei um die Funktion printf richtig zu übersetzen
#include <stdlib.h> //bindet die header-Datei für den befehl EXIT_SUCESS

int main(void) { /*hier beginnt das Programm, main ist eine Funktion die in jedem Programm  enthalten sein muss, dies ist der Einstiegspunkt in     einem Programm; void besagt das keine Datei übergeben wird, somit wird etwas zurückgegeben*/
    printf("Hello World!\n"); /*die Funktion printf übergibt die Zeichenkette "Hello World!"; der Befehl \n erzwingt einen Zeilenumbruch*/
    return EXIT_SUCCESS; // sorgt für die Rückgabe eines Wertes
}                       //dies wird alles in einer geschwungenen Klammer eingetragen

Aufgabe 7:
w - steht für "warnings"
-Wall schaltet alle Compiler Warning Flags an
-werror behandelt alle Fehler als Kompilierungsfehler
-Wextra meldet alle Warnungen oder Fehler, die während der Programmkompilierung auftreten
-wpendatic zeigt alle Warnungen an die nicht mit dem ISO C standard konform sind

-std=-c11 steht spezifiziert die ISO standard Version in ISO++11
-o s01e04 gibt den Output Namen an
